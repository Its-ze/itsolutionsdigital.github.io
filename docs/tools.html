// Tab switching
function showTab(tab) {
  document.querySelectorAll('.tab-content').forEach(e => e.style.display = 'none');
  document.querySelectorAll('.tab-btn').forEach(e => e.classList.remove('active'));
  document.getElementById(tab).style.display = 'block';
  document.querySelector('.tab-btn[onclick="showTab(\''+tab+'\')"]').classList.add('active');
}

// Helper: Validate domain
function isValidDomain(domain) {
  return /^[a-zA-Z0-9.-]+\.[a-zA-Z]{2,}$/.test(domain);
}
// Helper: Validate IP
function isValidIP(ip) {
  return /^((25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$/.test(ip);
}
// Copy result to clipboard
function copyResult(id) {
  const el = document.getElementById(id);
  if (el && el.textContent) {
    navigator.clipboard.writeText(el.textContent);
    el.style.background = '#222';
    setTimeout(()=>el.style.background='', 400);
  }
}

async function lookupDNSCombo() {
  const u = document.getElementById('dnscomboinput').value.trim(), r = document.getElementById('dnscomboresult');
  const type = document.getElementById('dnsrecordtype').value;
  if (!isValidDomain(u)) { r.textContent = 'âŒ Please enter a valid domain.'; return; }
  r.textContent = 'Looking up...';
  try {
    let url = 'https://dns.google/resolve?name=' + u + '&type=' + (type === 'SPF' ? 'TXT' : type);
    const res = await fetch(url);
    const data = await res.json();
    if (type === 'SPF') {
      if (data.Answer) {
        const spf = data.Answer.map(a => a.data).find(txt => txt.includes('v=spf1'));
        r.textContent = spf ? 'SPF Record:\n' + spf : 'âŒ No SPF record found.';
      } else {
        r.textContent = 'âŒ No TXT records.';
      }
    } else if (type === 'A' || type === 'AAAA' || type === 'MX' || type === 'CNAME') {
      if (data.Answer) {
        let label = type + ' Record' + (type === 'MX' ? 's' : '');
        r.textContent = label + ':\n' + data.Answer.map(a => a.data).join('\n');
      } else {
        r.textContent = 'âŒ No ' + type + ' records.';
      }
    } else {
      r.textContent = 'âŒ Unsupported record type.';
    }
  } catch {
    r.textContent = 'âŒ DNS query failed.';
  }
}

// SSL Info (using public API)
async function lookupSSL() {
  const u = document.getElementById('sslurl').value.trim(), r = document.getElementById('sslresult');

    const data = await res.json();
    if (data.endpoints && data.endpoints.length > 0) {
      let info = 'SSL Info:\n';
      data.endpoints.forEach(ep => {
        info += `IP: ${ep.ipAddress}\nGrade: ${ep.grade || 'N/A'}\n`; });
      r.textContent = info;
    } else {
      r.textContent = 'âŒ No SSL info.';
    }
  } catch {
    r.textContent = 'âŒ SSL info lookup failed.';
  }
}
